async function initializeApp() {
  const handler = {
    onDataChanged(data) {
      // Update UI if needed
    }
  };

  const result = await window.dataSdk.init(handler);
  if (!result.isOk) {
    console.error("Data SDK initialization failed");
  }

  // Element SDK setup
  const defaultConfig = {
    dashboard_title: 'KETSUEKI-GATA PERSONALITY TEST',
    subject_name: 'PSYCHOLOGICAL ANALYSIS',
    mbti_type: 'INTRJ',
    background_color: '#0a0a0f',
    surface_color: '#12121a',
    text_color: '#e0e0f0',
    primary_color: '#00f0ff',
    secondary_color: '#a855f7',
    font_family: 'Rajdhani',
    font_size: 16
  };

  window.elementSdk.init({
    defaultConfig,
    onConfigChange: async (config) => { applyConfig(config); },
    mapToCapabilities: (config) => ({
      recolorables: [
        { get: () => config.background_color || defaultConfig.background_color, set: (v) => { config.background_color = v; window.elementSdk.setConfig({ background_color: v }); } },
        { get: () => config.surface_color || defaultConfig.surface_color, set: (v) => { config.surface_color = v; window.elementSdk.setConfig({ surface_color: v }); } },
        { get: () => config.text_color || defaultConfig.text_color, set: (v) => { config.text_color = v; window.elementSdk.setConfig({ text_color: v }); } },
        { get: () => config.primary_color || defaultConfig.primary_color, set: (v) => { config.primary_color = v; window.elementSdk.setConfig({ primary_color: v }); } },
        { get: () => config.secondary_color || defaultConfig.secondary_color, set: (v) => { config.secondary_color = v; window.elementSdk.setConfig({ secondary_color: v }); } }
      ],
      borderables: [],
      fontEditable: {
        get: () => config.font_family || defaultConfig.font_family,
        set: (v) => { config.font_family = v; window.elementSdk.setConfig({ font_family: v }); }
      },
      fontSizeable: {
        get: () => config.font_size || defaultConfig.font_size,
        set: (v) => { config.font_size = v; window.elementSdk.setConfig({ font_size: v }); }

    }),
    mapToEditPanelValues: (config) => new Map([
      ['dashboard_title', config.dashboard_title || defaultConfig.dashboard_title],
      ['subject_name', config.subject_name || defaultConfig.subject_name],

  renderIntro();

  document.documentElement.style.setProperty('--cyan', config.primary_color);
  document.documentElement.style.setProperty('--violet', config.secondary_color);
  document.body.style.background = config.background_color;
  document.body.style.color = config.text_color;
}

function startTest() {
  currentScreen = 'test';
  currentQuestion = 0;
  answers = {};
  scores = { A: 0, B: 0, O: 0, AB: 0 };
  renderQuestion();
}

function renderQuestion() {
  if (currentQuestion >= testQuestions.length) {
    calculateResults();
    return;
  }

  const q = testQuestions[currentQuestion];
  const progress = ((currentQuestion) / testQuestions.length) * 100;

  const main = document.getElementById('mainContent');
  main.innerHTML = `
    <div class="test-container">
      <div class="progress-bar" style="margin-bottom: 30px;">
        <div class="progress-fill" style="width: ${progress}%"></div>
      </div>
      <p class="progress-text">PERTANYAAN ${currentQuestion + 1} DARI ${testQuestions.length} • PAKET ${q.package}</p>
      <div class="question-card" style="max-width: 900px; margin: 0 auto;">
        <h2 class="question-text">${q.question}</h2>
        <div class="answers-grid">
          ${q.answers.map((ans, idx) => `
            <div class="answer-card" onclick="selectAnswer(${idx})">
              <span class="answer-emoji">${ans.emoji}</span>
              <p class="answer-text">${ans.text}</p>
            </div>
          `).join('')}
        </div>
        <button class="btn btn-next" id="nextBtn" onclick="nextQuestion()" disabled>LANJUT KE PERTANYAAN BERIKUTNYA</button>
      </div>
    </div>
  `;
}

function selectAnswer(idx) {
  const q = testQuestions[currentQuestion];
  const ans = q.answers[idx];

  // Update scores
  scores[ans.type] += ans.weight;
  answers[currentQuestion] = idx;

  // Visual feedback
  document.querySelectorAll('.answer-card').forEach(card => card.classList.remove('selected'));
  document.querySelectorAll('.answer-card')[idx].classList.add('selected');

  // Enable next button
  const nextBtn = document.getElementById('nextBtn');
  nextBtn.classList.add('active');
  nextBtn.disabled = false;

  // Auto-advance after 600ms
  setTimeout(() => nextQuestion(), 600);
}

function nextQuestion() {
  currentQuestion++;
  renderQuestion();
}

function calculateResults() {
  // Determine dominant type
  let dominantType = 'A';
  let maxScore = scores.A;

  for (let type of ['B', 'O', 'AB']) {
    if (scores[type] > maxScore) {
      maxScore = scores[type];
      dominantType = type;
    }
  }

  currentScreen = 'results';
  renderResults(dominantType);
}

async function saveResult(dominantType) {
  const now = new Date();
  const total = scores.A + scores.B + scores.O + scores.AB;

  const result = await window.dataSdk.create({
    test_id: 'test_' + Date.now(),
    name: dominantType,
    completed_date: now.toISOString(),
    type_a: Math.round((scores.A / total) * 100),
    type_b: Math.round((scores.B / total) * 100),
    type_o: Math.round((scores.O / total) * 100),
    type_ab: Math.round((scores.AB / total) * 100),
    dominant_type: dominantType,
    result_text: typeDescriptions[dominantType].description
  });

  if (result.isOk) {
    alert('Hasil tes berhasil disimpan ke Canva Sheet Anda!');
  } else {
    console.error('Save failed', result.error);
  }
}

function createParticles() {

    p.style.top = Math.random() * 100 + '%';
    p.style.opacity = (Math.random() * 0.3 + 0.1).toString();
    p.style.animation = `particleDrift ${8 + Math.random() * 12}s ease-in-out infinite`;
    p.style.animationDelay = `-${Math.random() * 10}s`;
    container.appendChild(p);
  }
}
